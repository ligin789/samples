import React, { useState } from 'react'

const App = () => {
  const [name,setName]=useState('')
  function clickAble(): void {
    setName("hello")
  }

  return (
    <div>
      <h1 style={{fontSize:"10px"}}>{name}</h1>
      <button onClick={clickAble}>clickAble</button>
    </div>
  )
}

export default App
