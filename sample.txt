
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 	
import Colors from '../../globals/Colors';
import './ExceptionDots.css'
import Styles from './Styles';
const ExceptionDots =(props:any)=> {
    const {outerDivClassName = '', innerDivClassName = '', dotsClassName= '', dotType= '', isSelected, dotClick}: {outerDivClassName: string, innerDivClassName: string, dotsClassName:string, dotType:string, isSelected:boolean, dotClick:any } = props;
    return (
        <div onClick={dotClick}>
            <span className={outerDivClassName !== '' ? outerDivClassName : 'outerCircle'} style={{ ...Styles.outerCircle, backgroundColor: isSelected? Colors.DOT_OUTER_BORDER_COLOR : 'transparent' }} >
                <span className={innerDivClassName !== '' ? innerDivClassName : 'transparentCircle'} style={Styles.transparentCircle}>
                    <span className={dotsClassName !== '' ? dotsClassName : 'innerCircle'} style={{ ...Styles.innerCircle, backgroundColor: dotType === 'delayed' ? Colors.DOT_DELAYED_COLOR : Colors.DOT_ON_TIME_COLOR }} />
                </span>
            </span>
        </div>
    );
  }
  
  export default ExceptionDots;
