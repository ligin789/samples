import { create } from "zustand";
import { v4 } from "uuid";
import type { FormElement } from "../types";

type State = {
  elements: FormElement[];
  selectedElement: string | null;
  fieldValues: Record<string, any>;

  addElement: (type: FormElement["type"], parentId?: string | null) => void;
  deleteElement:(id:string)=>void;
  updateElement: (id: string, newProps: Partial<FormElement["props"]>) => void;
  selectElement: (id: string) => void;
  reorderElements: (newEls: FormElement[]) => void;
  updateFieldValue: (id: string, val: any) => void;
};

export const useFormStore = create<State>((set) => ({
  elements: [],
  selectedElement: null,
  fieldValues: {},

  deleteElement : (id) => {
    set((s)=>{
      const remove = (node: FormElement[]) : FormElement[] =>
        node.filter((el)=>el.id !== id)
            .map((el)=>
              "children" in el ? {...el,children:remove(el.children)} as FormElement : el
            )
        return{
          elements:remove(s.elements),
          selectedElement:null
        }
    })
  },

  addElement: (type, parentId = null) => {
    const newEl: any = { id: v4(), type, props: { style: {} } };
    if (type === "div") {
      newEl.children = [];
    }
    // newEl.props.style.height = '50px'
    if (type === "text") {
      newEl.props.content = 'label'
    }
    set((s) => {
      console.log("addElement called", { type, parentId, elements: s.elements });
      let updatedElements;
      if (!parentId) {
        updatedElements = [...s.elements, newEl];
      } else {
        const insert = (nodes: FormElement[]): FormElement[] =>
          nodes.map((el) => {
            if (el.id === parentId && el.type === "div") {
              return { ...el, children: [...el.children, newEl] } as FormElement;
            }
            if ("children" in el) {
              return { ...el, children: insert(el.children) } as FormElement;
            }
            return el;
          });
        updatedElements = insert(s.elements);
      }
      return { elements: updatedElements, selectedElement: newEl.id };
    });
  },

  updateElement: (id, newProps) =>
    set((s) => {
      const recurse = (nodes: FormElement[]): FormElement[] =>
        nodes.map((el) => {
          if (el.id === id) return { ...el, props: { ...el.props, ...newProps } };
          if ("children" in el) return { ...el, children: recurse(el.children) };
          return el;
        });
      return { elements: recurse(s.elements) };
  }),

  selectElement: (id) => set((prev)=>({ selectedElement: prev.selectedElement !==id ?id :null })),

  reorderElements: (newEls) => set({ elements: newEls }),

  updateFieldValue: (id, val) =>
    set((s) => ({ fieldValues: { ...s.fieldValues, [id]: val } })),
}));

      const internalRef = useRef<HTMLDivElement>(null);

    addElement(item.type, selectedElement);
  }, [addElement, selectedElement]);

    accept: ItemTypes.FORM_ELEMENT,
    drop: handleDrop,
    collect: (monitor) => ({
      isOver: !!monitor.isOver(),

  const  {setNodeRef} = useDroppable({
    id:'canvas-drop-area'

      const newIdx = elements.findIndex((e) => e.id === over.id);
      reorder(arrayMove(elements, oldIdx, newIdx));

import React, { useState } from "react";
import type { FormElement } from "../types";

interface Props {
  elements: FormElement[];
}

const FormPreview: React.FC<Props> = ({ elements }) => {
  const [data, setData] = useState<Record<string, any>>({});
  const handleChange = (id: string, val: any) => setData((d) => ({ ...d, [id]: val }));

  const renderEl = (el: FormElement): React.ReactNode => {
    const style = el.props.style;
    switch (el.type) {
      case "text":
        return <div key={el.id} style={style}>{el.props.content}</div>;
      case "text_input":
        return (
          <input
            key={el.id}
            style={style}
            placeholder={el.props.placeholder}
            onChange={(e) => handleChange(el.id, e.target.value)}
            className="border px-2 py-1 mb-2 w-full"
          />
        );
      case "div":
        return <div key={el.id} style={style} className="border p-2 mb-2">{el.children.map(renderEl)}</div>;
      case "checkbox":
      case "radio":
        return (
          <label key={el.id} style={style} className="flex items-center mb-2">
            <input type={el.type} onChange={(e) => handleChange(el.id, e.target.checked)} className="mr-2" />
            {el.props.label}
          </label>
        );
      case "switch":
        return (
          <label key={el.id} style={style} className="flex items-center mb-2">
            <input type="checkbox" onChange={(e) => handleChange(el.id, e.target.checked)} className="mr-2" />
            Switch
          </label>
        );
      case "image":
        return <img key={el.id} src={el.props.src} alt="" style={style} className="mb-2" />;
      default:
        return null;
    }
  };

  const onSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    alert(JSON.stringify(data, null, 2));
  };

  return (
    <form onSubmit={onSubmit} className="space-y-2">
      {elements.map(renderEl)}
      <button type="submit" className="bg-blue-600 text-white px-4 py-2 rounded">Submit</button>
    </form>
  );
};

export default FormPreview;


SideBar.tsx

import React from "react";
import { useDraggable } from "@dnd-kit/core";
import { useFormStore } from "../store/useFormStore";
import { useDrag } from 'react-dnd';
import { ItemTypes, FormElementType } from './Types';

const ELEMENTS= [{ type: "text", label: "Text" },
  { type: "div", label: "Container" },
  { type: "image", label: "Image" },
  { type: "text_input", label: "Text Input" },
  { type: "radio", label: "Radio" },
  { type: "checkbox", label: "Checkbox" },
  { type: "switch", label: "Switch" }] as const;

interface DraggableItemProps {
  type: FormElementType;
  selectedId: string | null;
  label:string |null;
}
const sidebarItemStyle: React.CSSProperties = {
  width: "100%",
  textAlign: "left",
  padding: "8px 12px",
  background: "#fff",
  marginBottom: "8px",
  borderRadius: "6px",
  boxShadow: "0 1px 2px rgba(0,0,0,0.04)",
  cursor: "grab",
  userSelect: "none",
  border: "1px solid #e5e7eb",
};

const sidebarItemHoverStyle: React.CSSProperties = {
  background: "#f3f4f6",
};

const DraggableItem: React.FC<DraggableItemProps> = ({ type, selectedId,label }) => {
  const add = useFormStore((s) => s.addElement);
  const {attributes, listeners, setNodeRef} = useDraggable({
    id:`sidebar-${type}`,
    data: {type}
  })
  const [{ isDragging }, drag] = useDrag(() => ({
    type: ItemTypes.FORM_ELEMENT,
    item: { type },
    collect: (monitor) => ({
      isDragging: !!monitor.isDragging(),
    }),
  }));

  const opacity = isDragging ? 0.4 : 1;

  return (
    <button
      ref={(node: HTMLButtonElement | null) => {
        if (node) {
          drag(node); // Connect the node to the drag source
        }
      }}
      onMouseOver={(e:any) => (e.currentTarget.style.background = sidebarItemHoverStyle.background!)}
      onMouseOut={(e:any) => (e.currentTarget.style.background = sidebarItemStyle.background!)}
      onClick={() => add(type, selectedId)}
      style={{ opacity ,...sidebarItemStyle }}
    >
      Add {type} {selectedId && "(inside selected)"}
    </button>
  );
};

export default function Sidebar() {
  const sel = useFormStore((s) => s.selectedElement);
  return (
    <div style={{
    width: "25%",
    padding: "16px",
    background: "#f3f4f6",
    minWidth: "180px",
    boxSizing: "border-box"
  }}>
      <h2 style={{ fontWeight: "bold", marginBottom: "8px" }}>Elements</h2>
      {ELEMENTS.map((el) => (
        <DraggableItem key={el.type} type={el.type} selectedId={sel} label={el.label} />
      ))}
    </div>
  );
}
