import Styles from "./Styles";

const MoreDetail = (props : any)=>{
    const {data, selectIndex, setSelectIndex, selectBox, domain,isAircraft} = props
    

    const handleSelectIndex = (index:any)=>{
        if(selectBox===index){
            return setSelectIndex(-1)
        }else{
            return setSelectIndex(index)
        }
    }
    const getDomain = ()=>{
        if(domain){
            if(domain==='Tail Connection')return tailConnectionGetDomain(data);
            if(domain==='crewInfo'){
                return crewInfoGetDomain(handleSelectIndex, selectIndex, data)
            }
            if(domain==='paxInfo'){
                return paxInfoGetDomain(handleSelectIndex, selectIndex, data)
            }
            if(domain==='text'){
                return textGetDomain(handleSelectIndex, selectIndex, data, isAircraft)
            }
        }else{
            return getDomainElse()
        }
    }
    return(
        <>
            {getDomain()}
        </>
    )
}
export default MoreDetail;

function getDomainElse() {
    return <div style={{ marginLeft: '25px', textAlign: 'left' }}>-</div>;
}

function textGetDomain(handleSelectIndex: (index: any) => any, selectIndex: any, data: any, isAircraft: any) {
    return <div style={{ ...Styles.moreDetailBoxDetail, ...{ textAlign: 'left', overflowX: 'auto' } }} onClick={() => { handleSelectIndex(selectIndex); } }>
        <div>{`${data} ${!isAircraft ? 'Pax' : ''}`}</div>
    </div>;
}

function paxInfoGetDomain(handleSelectIndex: (index: any) => any, selectIndex: any, data: any) {
    return <div style={{ ...Styles.moreDetailBoxDetail, ...{ textAlign: 'left', overflowX: 'auto' } }} onClick={() => { handleSelectIndex(selectIndex); } }>
        {data && data.length > 0 && data.map((el: any) => { return (<div key={el.paxCabinType}>{`${el.paxCabinType ? el.paxCabinType : '-'}`}</div>); })}
    </div>;
}

function crewInfoGetDomain(handleSelectIndex: (index: any) => any, selectIndex: any, data: any) {
    return <div style={{ ...Styles.moreDetailBoxDetail, ...{ textAlign: 'left', overflowX: 'auto' } }} onClick={() => { handleSelectIndex(selectIndex); } }>
        {data && data.length > 0 && data.map((el: any) => { return (<div key={el.staffId}>{`${el.staffId ? el.staffId : '-'} | ${el.name ? el.name : '-'} | ${el.crewWorkPosition ? el.crewWorkPosition : '-'}`}</div>); })}
    </div>;
}

function tailConnectionGetDomain(data: any) {
    {
        return <div style={{ marginLeft: '25px', textAlign: 'left' }}>{data}</div>;
    }
}
