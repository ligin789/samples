import React, { useState } from "react";
import "./index.css"
import { t } from "i18next";
 
export const HyperLinkPopup = (props: any) => {
  const { flightNoValue, setCriticalPopupVisible, setFlightLinkModel, setFlightFeedClick, setCreateAlertClicked, carrierCodeValue } = props
 
  const handleMarkCriticalHandler = () => {
    setCriticalPopupVisible(true)
    setFlightLinkModel(false)
  }
 
  const handleFlightFeedHandler = () => {
    setFlightFeedClick(true)
    setFlightLinkModel(false)
  }
 
  const createAlertHandler = () => {
    setCreateAlertClicked(true)
    setFlightLinkModel(false)
  }
 
  return (
    <div style={{ position: "absolute", zIndex: 10, right: "20%", top: "50%" }} >
      <div className="hyperlink-header">
        <span>
          <label className="headerLabel" style={{ paddingRight: "3px" }}>{carrierCodeValue}</label>
          <label className="headerLabel">{flightNoValue}</label>
        </span>
        <img src={require("../../assets/images/closeButton.png")} style={{ height: "29px", width: "29px" }} onClick={() => setFlightLinkModel(false)} />
      </div>
      <div className="popup-Container" >
        <div className="popup-label" onClick={createAlertHandler} >{t("createAlert")}</div>
        <div className="popup-label" onClick={handleMarkCriticalHandler}>{t("markCritical")}</div>
        <div className="popup-label" onClick={handleFlightFeedHandler}>{t("flightFeed")}</div>
        <div className="popup-label">{t("addFlightCard")}</div>
        <div className="popup-label">{t("addGateCard")}</div>
        <div className="popup-label-crew">{t("Crew")}</div>
      </div>
    </div>
 
 
  )
}
 
export default HyperLinkPopup
