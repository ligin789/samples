  <div class="current-time" id="currentTime">Time: 16:00</div>
 
  <script>
  const slider = document.getElementById('slider');
  const timeline = document.getElementById('timeline');
  const currentTime = document.getElementById('currentTime');
 
  const times = ["16:00", "17:00", "18:00", "19:00", "20:00", "21:00", "22:00", "23:00", "00:00"];
 
  let isDragging = false;
 
  timeline.addEventListener('mousedown', startDrag);
  window.addEventListener('mousemove', onDrag);
  window.addEventListener('mouseup', endDrag);
 
  function startDrag(e) {
    isDragging = true;
    moveSlider(e);
  }
 
  function onDrag(e) {
    if (isDragging) moveSlider(e);
  }
 
  function endDrag(e) {
    if (!isDragging) return;
    isDragging = false;
 
    const rect = timeline.getBoundingClientRect();
    let x = e.clientX - rect.left;
 
    // Clamp and snap to nearest index
    x = Math.max(0, Math.min(x, rect.width));
    const percent = x / rect.width;
    const index = Math.round(percent * (times.length - 1));
 
    const snappedX = (index / (times.length - 1)) * rect.width;
slider.style.left = `${snappedX - 0}px`;
 
    currentTime.textContent = `Time: ${times[index]}`;
  }
 
  function moveSlider(e) {
    const rect = timeline.getBoundingClientRect();
    let x = e.clientX - rect.left;
 
    x = Math.max(0, Math.min(x, rect.width));
slider.style.left = `${x - 10}px`;
 
    const percent = x / rect.width;
    const index = Math.floor(percent * (times.length - 1));
    const nextIndex = Math.min(index + 1, times.length - 1);
 
    const baseTime = times[index];
    const nextTime = times[nextIndex];
 
    const baseMinutes = toMinutes(baseTime);
    const nextMinutes = toMinutes(nextTime);
    const currentMinutes = baseMinutes + (nextMinutes - baseMinutes) * ((percent * (times.length - 1)) % 1);
 
    currentTime.textContent = "Time: " + fromMinutes(currentMinutes);
  }
 
  function toMinutes(timeStr) {
    const [h, m] = timeStr.split(":").map(Number);
    return h * 60 + m;
  }
 
  function fromMinutes(mins) {
    const h = Math.floor(mins / 60) % 24;
    const m = Math.round(mins % 60);
    return `${h.toString().padStart(2, '0')}:${m.toString().padStart(2, '0')}`;
  }
</script>
