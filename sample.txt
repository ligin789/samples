import moment from 'moment';
import FlightIcon from '../../assets/ganttFlightIcon.svg'
import React from 'react';
import MouseOverPopover from '../mousePopOver';
import './flightSlot.css';
import Styles from './Styles';

const FlightSlot = (props: any) => {
    const { otherStartTime, configLine, type, track, detailData, slotWidth, selectedFlight } = props;
    detailData.arrFlightNo === 'UA 470' && console.log(otherStartTime, detailData.ETD, detailData.STD, 'check FlightSlot FlightSlot props ==>', props, props?.detailData?.arrFlightNo?.split(" ")[1] === selectedFlight)
    detailData.arrFlightNo === 'UA 1528' && console.log(otherStartTime,'check for current main box => ',detailData)
    React.useEffect(() => {
        if (detailData?.arrFlightNo?.split(" ")[1] === selectedFlight || detailData?.depFlightNo?.split(" ")[1] === selectedFlight) {
            document.getElementById(selectedFlight)?.scrollIntoView({ block: "center", inline: "nearest" });
        }
    }, [selectedFlight]);

    const getCurrentInFlightBoxWidth = ()=>{
        let mainBarConst = detailData?.mainBar > 0 ? 1 : -1
        if(detailData.tarmecIn > (mainBarConst * detailData?.mainBar)){
            return 0
        }else{
            return detailData.tarmecIn
        }
    }

    const getCurrentSlotDetailTimeBoxMarginTop = ()=>{
        let tempWidth = detailData.subMainBar > detailData.mainBar ? detailData.subMainBar : detailData.mainBar
        if(track * tempWidth > 330 ){
            return 20
        }else{
            return 0
        }
    }

    const getCurrentSlotGanttMain = ()=>{
        let tempTrack = track
        let mainBarConst = detailData?.mainBar > 0 ? 1 : -1
        if(detailData.currentMainBar < 0 ){
            return tempTrack*0
        }else{
            if(detailData.currentMainBar > (mainBarConst * detailData?.mainBar)){
                return tempTrack*0
            }else{
                return tempTrack*detailData.currentMainBar
            }
        }
    }

    const getCurrentSlotGanttMainBox = ()=>{
        if(detailData.subMainBreakDown.arrDelay < 0){
            if(moment(otherStartTime,'YYYY-MM-DDTHH:mm:SS').valueOf()>=moment(detailData?.ETA,'YYYY-MM-DDTHH:mm:SS').valueOf()){
                return 0
            }else{
                return ((track) * (-detailData.subMainBreakDown.arrDelay))
            }
        }else{return 0} 
    }

    const getCurrentSlotGanttSubMain = ()=>{
        let tempTrack = track
        let tempMain = detailData.mainBar
        let mainBarConst = detailData?.mainBar > 0 ? 1 : -1
        if(detailData.currentMainBar < 0){
            return tempTrack*(tempMain-0)
        }else{
            if(detailData.currentMainBar > (mainBarConst * detailData?.mainBar)){
                return tempTrack*(tempMain-0)
            }else{
                return tempTrack*(tempMain-detailData.currentMainBar)
            }
        }
    }

    const getOtherSlotBox = ()=>{
        let final = 0
        let tempTrack = track
        let tempArrCount = detailData?.subMainBreakDown?.arrDelay > 0 ? 1 : -1

        if(detailData.subMainBar > detailData.mainBar){
            final = final+(tempTrack*detailData.subMainBar)
        }else{
            final = final+(tempTrack*detailData.mainBar)
        }

        if(detailData?.subMainBreakDown?.arrDelay === 0){
            final = final+(tempTrack*0)
        }else{
            final = final+(tempTrack*(tempArrCount * detailData?.subMainBreakDown?.arrDelay))
        }

        return final
    }

    const getOtherSlotGanttSubBox = ()=>{
        let final = 0
        let tempTrack = track
        let tempArrCount = detailData?.subMainBreakDown?.arrDelay > 0 ? 1 : -1
        if(detailData.subMainBar > 0){
            final = final+(tempTrack*detailData.subMainBar)
        }else{
            final = final+(tempTrack*detailData.mainBar)
        } 
        if(detailData?.subMainBreakDown?.arrDelay === 0){
            final = final+(tempTrack*0)
        }else{
            final = final+(tempTrack*(tempArrCount * detailData?.subMainBreakDown?.arrDelay))
        }
         
        return final
    }

    const getOtherSlotGanttMain = ()=>{
        if(detailData?.subMainBreakDown?.arrDelay < 0){
            return `${((track) * (-1 * detailData?.subMainBreakDown?.arrDelay))}`
        }else{
            return ''
        }
    }

    const getOtherSlotGanttBox = ()=>{
        let finalData = 0
        let tempTrack = track
        let tempArrCount = detailData?.subMainBreakDown?.arrDelay > 0 ? 1 : -1

        if(detailData.subMainBar > 0){
            finalData = finalData+(tempTrack*detailData.subMainBar)
        }else{ 
            finalData = finalData+(tempTrack*detailData.mainBar)
        }

        if(detailData?.subMainBreakDown?.arrDelay === 0){
            finalData = finalData+(tempTrack*0)
        }else{
            finalData = finalData+(tempTrack*(tempArrCount * detailData?.subMainBreakDown?.arrDelay))
        }

        return finalData
    }

    const getPastSlotBox = ()=>{
        let tempArrCount = detailData?.subMainBreakDown?.arrDelay >= 0 ? 0 : -1
        if(Date.UTC(Number(otherStartTime?.split('T')[0]?.split('-')[0]), Number(otherStartTime?.split('T')[0]?.split('-')[1]), Number(otherStartTime?.split('T')[0]?.split('-')[2]), Number(otherStartTime?.split('T')[1]?.split(':')[0]), Number(otherStartTime?.split('T')[1]?.split(':')[1])) > Date.UTC(Number(detailData?.ETA?.split('T')[0]?.split('-')[0]), Number(detailData?.ETA?.split('T')[0]?.split('-')[1]), Number(detailData?.ETA?.split('T')[0]?.split('-')[2]), Number(detailData?.ETA?.split('T')[1]?.split(':')[0]), Number(detailData?.ETA?.split('T')[1]?.split(':')[1]))){
            return '0px'
        }else{
            return `${((track) * detailData.subMainBar) + ((track) * (tempArrCount * detailData?.subMainBreakDown?.arrDelay))}px`
        }
    }

    const getPastSlotDetailTimeBox = ()=>{
        let tempCheck = detailData.subMainBar > detailData.mainBar ? detailData.subMainBar : detailData.mainBar
        if(track * tempCheck > 240){
            return 20
        }else{
            return 0
        }
    }

    const getCurrentFligh = ()=>{
        return (
            <div className='currentMainBox' style={{ width: `${slotWidth}px` }}>
                <div className='currentInFlinghtBox' style={{ width: `${(track) * getCurrentInFlightBoxWidth()}px` }}>
                    <div style={{width:'0px'}}><div className='currentInFlinghtIconBox' style={{ width: `${(track) * getCurrentInFlightBoxWidth()}px` }}>
                        <img src={FlightIcon} className='currentInFlinghtIcon' />
                        <div className='currentInFlinghtIconLine' style={{...Styles.currentFlightSlotPrimeColorBG,...{ width: `${(track) * getCurrentInFlightBoxWidth()}px` }}}></div>
                    </div></div>
                    <div className='currentInFlinghtLine' style={{...Styles.flightLineBG,...{ width: `${(track) * getCurrentInFlightBoxWidth()}px` }}}></div>
                </div>
                <div className='currentSlotBox' style={{ width: `${(track) * (detailData.subMainBar > detailData.mainBar ? detailData.subMainBar : detailData.mainBar)}px` }}>

                    <div style={{width:'0px'}}><div className='currentSlotDetailBox' style={{ font: 'normal normal normal 15px / 20px Roboto', width: `${(track) * (detailData.subMainBar > detailData.mainBar ? detailData.subMainBar : detailData.mainBar)}px` }}>
                        <div className='currentSlotDetailDataBox' style={{ width: `${(track) * (detailData.subMainBar > detailData.mainBar ? detailData.subMainBar : detailData.mainBar)}px` }}>
                            <div className='currentSlotDetailDataInFlightBox'>
                                <div className='currentSlotDetailDataInFlightStn'>{detailData.arr}</div>
                                <div className='currentSlotDetailDataInFlightFlt' style={Styles.flightSlotSubText}>{detailData.arrFlightNo}</div>
                            </div>
                            <div><MouseOverPopover configLine={configLine} conflictBox={detailData?.conflictBox} value={detailData.popUpDetail} /></div>
                            <div className='currentSlotDetailDataOutFlightBox'>
                                <div className='currentSlotDetailDataOutFlightFlt' style={Styles.flightSlotSubText}>{detailData.depFlightNo}</div>
                                <div className='currentSlotDetailDataOutFlightStn'>{detailData.dep}</div>
                            </div>
                        </div>
                        <div className='currentSlotDetailTimeBox' style={{ width: `${(track) * (detailData.subMainBar > detailData.mainBar ? detailData.subMainBar : detailData.mainBar)}px`, marginTop: `${getCurrentSlotDetailTimeBoxMarginTop()}px` }}>
                            <div className='currentSlotDetailTimeInFlightBox'>
                                <div>{moment(detailData.ETA?.substring(0, 19)).format('HH:mm')}</div>
                                <div style={Styles.flightSlotSubText}>&nbsp;{`(${moment(detailData.STA?.substring(0, 19)).format('HH:mm')})`}</div>
                            </div>
                            <div className='currentSlotDetailTimeOutFlightBox'>
                                <div>&nbsp;{moment(detailData.ETD?.substring(0, 19)).format('HH:mm')}</div>
                                <div style={Styles.flightSlotSubText}>&nbsp;{`(${moment(detailData.STD?.substring(0, 19)).format('HH:mm')})`}</div>
                            </div>
                        </div>
                    </div></div>
                    <div id={detailData?.arrFlightNo?.split(" ")[1]} style={{...(detailData?.arrFlightNo?.split(" ")[1] === selectedFlight||detailData?.depFlightNo?.split(" ")[1]===selectedFlight)&&Styles.flightSlotForCurrentAndUpComing,...{ width: `${(track) * (detailData.subMainBar > detailData.mainBar ? detailData.subMainBar : detailData.mainBar)}px` }}} className={(detailData?.arrFlightNo?.split(" ")[1] === selectedFlight || detailData?.depFlightNo?.split(" ")[1]===selectedFlight) ? 'currentSlotGanttBox':''}>
                        <div className='currentSlotGanttMainBox' style={{...Styles.currentFlightSlotPrimeColorBG,...Styles.currentFlightSlotPrimeColorBorder,...{ width: `${(track) * detailData.mainBar}px`, marginLeft: `${getCurrentSlotGanttMainBox()}px` }}}>

                            <div className='currentSlotGanttMain' style={{...Styles.currentFlightSlotPrimeColorBG,...{ width: `${getCurrentSlotGanttMain()}px` }}}></div>
                            <div className='currentSlotGanttSubMain' style={{ width: `${getCurrentSlotGanttSubMain()}px` }}></div>
                        </div>
                        <div style={{width:'0px',marginTop:'-10px'}}>
                        <div className='currentSlotGanttCurrentTimeBox'>
                            <div className='currentSlotGanttCurrentTimeArrDealyBox' style={{ width: `${(track) * detailData.subMainBreakDown.arrDelay}px`, border: detailData.subMainBreakDown.arrDelay > 0 ? '1px solid #3483F4' : '' }}>
                                <div className='currentSlotGanttCurrentTimeArrDealy' style={{ width: `${(track) * detailData.subMainBreakDown.arrDelay}px` }}>
                                    <div className='currentSlotGanttCurrentTimeArrDealyLine' style={{...Styles.currentFlightSlotPrimeColorBG,...{ width: `${(track) * detailData.subMainBreakDown.arrDelay}px` }}}></div>
                                </div>
                            </div>

                            <div className='currentSlotGanttCurrentTimeMainBox' style={{ width: `${(track) * (detailData.subMainBar)}px` }}>
                                <div className='currentSlotGanttCurrentTimeMain' style={{...Styles.currentFlightSlotPrimeColorBG,...Styles.currentFlightSlotPrimeColorBorder,...{ width: `${(track) * (detailData.subMainBreakDown.currentRemain)}px` }}}></div>
                                <div className='currentSlotGanttCurrentTimeSubMain' style={{ width: `${(track) * (detailData.subMainBreakDown.otherRemain)}px` }}></div>
                            </div>
                        </div>
                        </div>
                    </div>
                </div>
                <div className='currentOutFlinghtBox' style={{ width: `${(track) * detailData.tarmecOut}px` }}>
                    <div style={{width:'0px'}}><div className='currentOutFlinghtIconBox' style={{ width: `${(track) * detailData.tarmecOut}px` }}>

                        <div className='currentOutFlinghtIconLine' style={{...Styles.currentFlightSlotPrimeColorBorderTopDashed,...{ width: `${(track) * detailData.tarmecOut}px` }}}></div>

                        <div style={{width:'0px'}}><img src={FlightIcon} className='currentOutFlinghtIcon' style={{ marginLeft: `${((track) * detailData.tarmecOut < 0 ? 0 : (track) * detailData.tarmecOut)}px` }} /></div>
                    </div></div>
                    <div className='currentOutFlinghtLine' style={{...Styles.flightLineBG,...{ width: `${(track) * detailData.tarmecOut}px` }}}></div>
                </div>
            </div>
        )
    }

    const getOtherFlight =()=>{
        return (
            <div className='otherMainBox' style={{ width: `${slotWidth}px` }}>
                <div className='otherInFlinghtBox' style={{ width: `${(track) * detailData.tarmecIn}px` }}>
                    <div className='otherInFlinghtLine' style={{...Styles.flightLineBG,...{ width: `${(track) * detailData.tarmecIn}px` }}}></div>
                </div>

                <div className='otherSlotBox' style={{ width: `${getOtherSlotBox()}px` }}>
                    <div style={{width:'0px'}}><div className='otherSlotDetailBox' style={{ font: 'normal normal normal 15px / 20px Roboto', width: `${slotWidth}px` }}>
                        <div className='otherSlotDetailDataBox' style={{ width: `${slotWidth}px` }}>
                            <div className='otherSlotDetailDataInFlightBox'>
                                <div className='otherSlotDetailDataInFlightStn'>{detailData.arr}</div>
                                <div className='otherSlotDetailDataInFlightFlt' style={Styles.flightSlotSubText}>{detailData.arrFlightNo}</div>
                            </div>

                            <div><MouseOverPopover configLine={configLine} conflictBox={detailData?.conflictBox} value={detailData.popUpDetail} /></div>
                            <div className='otherSlotDetailDataOutFlightBox'>
                                <div className='otherSlotDetailDataOutFlightFlt' style={Styles.flightSlotSubText}>{detailData.depFlightNo}</div>
                                <div className='otherSlotDetailDataOutFlightStn'>{detailData.dep}</div>
                            </div>
                        </div>
                        <div className='otherSlotDetailTimeBox' style={{ width: `${slotWidth}px`, marginTop: `${slotWidth > 330 ? 20 : 0}px` }}>
                            <div className='otherSlotDetailInFlightTimeBox'>
                                <div>{moment(detailData.ETA?.substring(0, 19)).format('HH:mm')}</div>
                                <div style={Styles.flightSlotSubText}>&nbsp;{`(${moment(detailData.STA?.substring(0, 19)).format('HH:mm')})`}</div>
                            </div>
                            <div className='otherSlotDetailOutFlightTimeBox'>
                                <div>&nbsp;{moment(detailData.ETD?.substring(0, 19)).format('HH:mm')}</div>
                                <div style={Styles.flightSlotSubText}>&nbsp;{`(${moment(detailData.STD?.substring(0, 19)).format('HH:mm')})`}</div>
                            </div>
                        </div>
                    </div></div>

                    <div id={detailData?.arrFlightNo?.split(" ")[1]} className={(detailData?.arrFlightNo?.split(" ")[1] === selectedFlight || detailData?.depFlightNo?.split(" ")[1]===selectedFlight) ?'otherSlotGanttBox':''} style={{...(detailData?.arrFlightNo?.split(" ")[1] === selectedFlight||detailData?.depFlightNo?.split(" ")[1]===selectedFlight)&&Styles.flightSlotForCurrentAndUpComing,...{ width: `${getOtherSlotGanttBox()}px`, height: '10px' }}}>
                        <div className='otherSlotGanttSubBox' 
                            style={{ width: `${getOtherSlotGanttSubBox()}px`}}
                        >
                            <div className='otherSlotGanttSubBoxArrDelay' 
                                style={{width: `${(track) * detailData.subMainBreakDown.arrDelay}px`}}
                            ></div>
                            <div className='otherSlotGanttSubBoxRemain' 
                                style={{width: `${(track) * detailData.subMainBreakDown.remain}px`}}
                            ></div>
                        </div>
                        <div className='otherSlotGanttMain' 
                            style={{ 
                                width: `${(track) * detailData.mainBar}px`, 
                                marginLeft: `${getOtherSlotGanttMain()}px`,
                            }}
                        ></div>
                    </div>
                </div>
                <div className='otherOutFlinghtBox' style={{ width: `${(track) * detailData.tarmecOut}px` }}>
                    <div className='otherOutFlinghtLine' style={{...Styles.flightLineBG,...{ width: `${(track) * detailData.tarmecOut}px` }}}></div>
                </div>
            </div>
        )
    }
    const getPastFlight = ()=>{
        return (
            <div className='pastMainBox' style={{ width: `${slotWidth}px` }}>
                <div className='pastInFlinghtBox' style={{ width: Date.UTC(Number(otherStartTime?.split('T')[0]?.split('-')[0]), Number(otherStartTime?.split('T')[0]?.split('-')[1]), Number(otherStartTime?.split('T')[0]?.split('-')[2]), Number(otherStartTime?.split('T')[1]?.split(':')[0]), Number(otherStartTime?.split('T')[1]?.split(':')[1])) > Date.UTC(Number(detailData?.STA?.split('T')[0]?.split('-')[0]), Number(detailData?.STA?.split('T')[0]?.split('-')[1]), Number(detailData?.STA?.split('T')[0]?.split('-')[2]), Number(detailData?.STA?.split('T')[1]?.split(':')[0]), Number(detailData?.STA?.split('T')[1]?.split(':')[1])) ? '0px' : `${(track) * detailData.tarmecIn}px` }}>
                    <div style={{width:'0px'}}><div className='pastInFlinghtIconBox' style={{ width:Date.UTC(Number(otherStartTime?.split('T')[0]?.split('-')[0]), Number(otherStartTime?.split('T')[0]?.split('-')[1]), Number(otherStartTime?.split('T')[0]?.split('-')[2]), Number(otherStartTime?.split('T')[1]?.split(':')[0]), Number(otherStartTime?.split('T')[1]?.split(':')[1])) > Date.UTC(Number(detailData?.STA?.split('T')[0]?.split('-')[0]), Number(detailData?.STA?.split('T')[0]?.split('-')[1]), Number(detailData?.STA?.split('T')[0]?.split('-')[2]), Number(detailData?.STA?.split('T')[1]?.split(':')[0]), Number(detailData?.STA?.split('T')[1]?.split(':')[1])) ? '0px' : `${(track) * detailData.tarmecIn}px` }}>
                        <img src={FlightIcon} className='pastInFlinghtIcon' />
                        <div className='pastInFlinghtIconLine' style={{...Styles.pastFlightSlotPrimeColorBG,...{width:Date.UTC(Number(otherStartTime?.split('T')[0]?.split('-')[0]), Number(otherStartTime?.split('T')[0]?.split('-')[1]), Number(otherStartTime?.split('T')[0]?.split('-')[2]), Number(otherStartTime?.split('T')[1]?.split(':')[0]), Number(otherStartTime?.split('T')[1]?.split(':')[1])) > Date.UTC(Number(detailData?.STA?.split('T')[0]?.split('-')[0]), Number(detailData?.STA?.split('T')[0]?.split('-')[1]), Number(detailData?.STA?.split('T')[0]?.split('-')[2]), Number(detailData?.STA?.split('T')[1]?.split(':')[0]), Number(detailData?.STA?.split('T')[1]?.split(':')[1])) ? '0px' : `${(track) * detailData.tarmecIn}px` }}}></div>
                    </div></div>
                    <div className='pastInFlinghtLine' style={{...Styles.pastFlightSlotPrimeColorBG,...{ width: Date.UTC(Number(otherStartTime?.split('T')[0]?.split('-')[0]), Number(otherStartTime?.split('T')[0]?.split('-')[1]), Number(otherStartTime?.split('T')[0]?.split('-')[2]), Number(otherStartTime?.split('T')[1]?.split(':')[0]), Number(otherStartTime?.split('T')[1]?.split(':')[1])) > Date.UTC(Number(detailData?.STA?.split('T')[0]?.split('-')[0]), Number(detailData?.STA?.split('T')[0]?.split('-')[1]), Number(detailData?.STA?.split('T')[0]?.split('-')[2]), Number(detailData?.STA?.split('T')[1]?.split(':')[0]), Number(detailData?.STA?.split('T')[1]?.split(':')[1])) ? '0px' : `${(track) * detailData.tarmecIn}px` }}}></div>
                </div>
                <div className='pastSlotBox' style={{ width: getPastSlotBox() }}>
                    <div style={{width:'0px'}}><div className='pastSlotDetailBox' style={{ font: 'normal normal normal 15px / 20px Roboto', width: `${(track) * (detailData.subMainBar > detailData.mainBar ? detailData.subMainBar : detailData.mainBar)}px` }}>
                        <div className='pastSlotDetailDataBox' style={{ width: `${(track) * (detailData.subMainBar > detailData.mainBar ? detailData.subMainBar : detailData.mainBar)}px` }}>
                            <div className='pastSlotDetailDataInFlightBox'>
                                <div className='pastSlotDetailDataInFlightStn'>{detailData.arr}</div>
                                <div className='pastSlotDetailDataInFlightFlt' style={Styles.flightSlotSubText}>{detailData.arrFlightNo}</div>
                            </div>
                            <div><MouseOverPopover configLine={configLine} conflictBox={detailData?.conflictBox} value={detailData.popUpDetail} /></div>
                            <div className='pastSlotDetailDataOutFlightBox'>
                                <div className='pastSlotDetailDataOutFlightFlt' style={Styles.flightSlotSubText}>{detailData.depFlightNo}</div>
                                <div className='pastSlotDetailDataOutFlightStn'>{detailData.dep}</div>
                            </div>
                        </div>
                        <div className='pastSlotDetailTimeBox' style={{ width: `${((track) * (detailData.subMainBar > detailData.mainBar ? detailData.subMainBar : detailData.mainBar))}px`, marginTop: `${getPastSlotDetailTimeBox()}px` }}>
                            <div className='pastSlotDetailTimeInFlightBox'>
                                <div>{moment(detailData.ETA?.substring(0, 19)).format('HH:mm')}</div>
                                <div style={Styles.flightSlotSubText}>&nbsp;{`(${moment(detailData.STA?.substring(0, 19)).format('HH:mm')})`}</div>
                            </div>
                            <div className='pastSlotDetailTimeOutFlightBox'>
                                <div>&nbsp;{moment(detailData.ETD?.substring(0, 19)).format('HH:mm')}</div>
                                <div style={Styles.flightSlotSubText}>&nbsp;{`(${moment(detailData.STD?.substring(0, 19)).format('HH:mm')})`}</div>
                            </div>
                        </div>
                    </div></div>
                    <div className='pastSlotGanttBox' style={{ width: `${(track) * detailData.subMainBar}px` }}>
                        <div className='pastSlotGanttMain' style={{...Styles.pastFlightSlotPrimeColorBG,...{ width: `${(track) * detailData.mainBar}px` }}}></div>
                        <div style={{width:'0px',marginTop:'-10px'}}>
                        <div className='pastSlotGanttSubBox' style={{...Styles.pastFlightSlotPrimeColorBG,...{ width: `${(track) * detailData.subMainBar}px` }}}>
                            <div className='pastSlotGanttSubArrDelayBox' style={{...Styles.pastFlightSlotPrimeColorBG,...Styles.pastFlightSlotPrimeColorBorder,...{ width: `${(track) * detailData.arrDelay}px` }}}>
                                <div className='pastSlotGanttSubArrDelay' style={{ width: `${(track) * detailData.arrDelay}px` }}></div>
                            </div>
                            <div className='pastSlotGanttSub' style={{...Styles.pastFlightSlotPrimeColorBG,...Styles.pastFlightSlotPrimeColorBorder,...{ width: `${(track) * detailData.remain}px` }}}></div>
                        </div>
                        </div>
                    </div>
                </div>
                <div className='pastOutFlinghtBox' style={{ width: `${(track) * detailData.tarmecOut}px` }}>
                    <div style={{width:'0px'}}><div className='pastOutFlinghtIconBox' style={{ width: `${(track) * detailData.tarmecOut}px` }}>
                        <div className='pastOutFlinghtIconLine' style={{...Styles.pastFlightSlotPrimeColorBG,...{ width: `${(track) * detailData.tarmecOut}px` }}}></div>
                        <div style={{width:'0px'}}><img src={FlightIcon} className='pastOutFlinghtIcon' style={{ marginLeft: `${(track) * detailData.tarmecOut}px` }} /></div>
                    </div></div>
                    <div className='pastOutFlinghtLine' style={{...Styles.flightLineBG,...{ width: `${(track) * detailData.tarmecOut}px` }}}></div>
                </div>
            </div>
        )
    }

    const getFlightGantt = ()=>{
        if(type === "current"){return (getCurrentFligh())}
        if(type === "other"){return (getOtherFlight())}
        if(type === "past"){return (getPastFlight())}
        if(type === "sameOther"){return (
            <div className='sameOtherMainBox' style={{ width: `${slotWidth}px` }}>
                <div className='sameOtherMainBoxLine' style={{...Styles.pastFlightSlotPrimeColorBG,...{ width: `${slotWidth}px` }}}></div>
            </div>
        )}
    }
    return (<div>
        {getFlightGantt()}
    </div>)
}

export default FlightSlot;
