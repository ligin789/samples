import moment from "moment"

const RenderDateAndTime = (props: any) => {
    const {isUTCToggled} = props;
    if (props.value === '—' || !props.value) {
      return <span>{'—'}</span>
    }
    
    return <span>{moment.utc(props.value).format('HH:mm')}</span>
}

export default RenderDateAndTime
