    function* getGanttDetailData({payload}: any) {

      let headerPayload: any = getPayloadWorkAndWidget()
      headerPayload.station = payload.station
      headerPayload.terminal = payload.terminal
      headerPayload.enterpriseCode = payload.enterpriseCode

    let response: CustomResponseType = yield APIcall(
      'POST',
      configuration.url.getganttdetails,
      headerPayload
    )
    console.log("getGanttDetailData",response);
    
    if (response.status === 200 || response.status === 304) {
        yield put({ 
          type: Constants.SAVE_GANTT_VIEW_DETAIL, 
          // payload: response.data 
          // below thing for testing purpose
          payload: GanttData
        })
    } else if (response?.response?.status === 401 || response?.response?.status === 403) {
      yield put({
        type: 'UPDATE_ALERT', payload: {
          title: 'Session Timeout',
          content: 'Login to start a new session'
        }
      })
    } else {
      yield put({
        type: 'UPDATE_ALERT', payload: {
          title: 'Alert',
          content: 'Oops, something went wrong. Please try again later.'
        }
      })
    }
  }

  function* callAdvisory({payload}: any) {
    try {
      let headerPayload: any = getPayloadCallAdvisory()
      headerPayload.station = payload.station
      headerPayload.terminal = payload.terminal
      headerPayload.enterpriseCode = payload.enterpriseCode
      let requestBody :any= headerPayload
      let tempId = Date.now().toString();
      requestBody["resolveProcessId"] = tempId
      requestBody["id"]= tempId
      requestBody["station"]=localStorage.getItem('selectedStation')
      localStorage.setItem('resolveProcessId',requestBody.resolveProcessId);
      console.log('check ne add of resoveProcessId ==> ',requestBody)
      let response: CustomResponseType = yield APIcall(
        'POST',
        configuration.url.mrvGateAdvisory,
        requestBody
      )
      console.log("callAdvisory",response);
      
      if (response?.data?.status === "200") {
          if(response?.data?.message!='Message posted'){
          localStorage.setItem('resolveProcessId','');
          yield put(
            {
              type:'UPDATE_ALERT', 
              payload:{
                title: 'Alert',
                content: `${response?.data?.message}`}
            }
          )
        }
      }
      else if (response?.data?.status === 401 || response?.data?.status === 403) {
        localStorage.setItem('resolveProcessId','');
        yield put({
          type: 'UPDATE_ALERT', payload: {
            title: 'Session Timeout',
            content: 'Login to start a new session'
          }
        })
      }
    } catch (error) {
      localStorage.setItem('resolveProcessId','');
      yield put({
        type: 'UPDATE_ALERT', payload: {
          title: 'Alert',
          content: 'Oops, something went wrong. Please try again later.'
        }
      })
    }
}

function* resolveProcessCheck({payload}: any) {
  if(localStorage.getItem('resolveProcessId')&&localStorage.getItem('resolveProcessId')!=''){
    try {
      let headerPayloadMain = {
        resolveProcessId:localStorage.getItem('resolveProcessId'),
        station:payload?.station ,
        terminal: payload?.terminal,
        enterpriseCode: payload?.enterpriseCode
      }
      let response: CustomResponseType = yield APIcall(
        'POST',
        configuration.url.resolveProcessCheck,
        // {
        //   resolveProcessId:localStorage.getItem('resolveProcessId'),
        //   station:`${localStorage.getItem('selectedStation')}`,
        //   terminal:`${localStorage.getItem('selectedTerminal')}`,
        //   enterpriseCode:`${localStorage.getItem('selectedEnterpriseCode')}`
        // }
        headerPayloadMain
      )
      console.log("resolveProcessCheck",response);
      
      if (response?.data?.status === "200") {
        let headerPayload = {
          station:payload?.station ,
          terminal: payload?.terminal,
          enterpriseCode: payload?.enterpriseCode
        }
        if(response?.data?.message==='Gate Advisory Posted Sucessfully'){
          localStorage.setItem('resolveProcessId','');
          yield put(flightCardViewAction.getFlightCardData(headerPayload));
        }

        if(response?.data?.message==='Gates are Not Advised'){
          localStorage.setItem('resolveProcessId','');
          yield put(
            {
              type:'UPDATE_ALERT', 
              payload:{
                title: 'Alert',
                content: `${response?.data?.message} for the requested conflict flights.`}
            }
          )
          yield put(flightCardViewAction.getFlightCardData(headerPayload));
        }
      }else{
        yield put({
          type: 'UPDATE_ALERT', payload: {
            title: 'Alert',
            content: 'Oops, something went wrong. Please try again later.'
          }
        })
      }
    } catch (error) {
      yield put({
        type: 'UPDATE_ALERT', payload: {
          title: 'Alert',
          content: 'Oops, something went wrong. Please try again later.'
        }
      })
    }
  }
}

export function* ganttViewDetailSaga(){
    yield takeLatest(Constants.GET_GANTT_VIEW_DETAIL,getGanttDetailData)
    yield takeLatest('CALL ADVISORY',callAdvisory)
    yield takeLatest('POST CONFIG DATA',postConfigData)
    yield takeLatest('POST CONFIG TYPE',postConfigType)
    yield takeLatest('POST CONFIG CHANGE',postConfigChange)
    yield takeLatest('GET RESOLVE PROCESS UPDATE',resolveProcessCheck)
}

function getDomainElse() {
    return <div style={{ marginLeft: '25px', textAlign: 'left' }}>-</div>;
}
 
function textGetDomain(handleSelectIndex: (index: any) => any, selectIndex: any, data: any, isAircraft: any) {
    return <div style={{ ...Styles.moreDetailBoxDetail, ...{ textAlign: 'left', overflowX: 'auto' } }} onClick={() => { handleSelectIndex(selectIndex); } }>
        <div>{`${data} ${!isAircraft ? 'Pax' : ''}`}</div>
    </div>;
}
 
function paxInfoGetDomain(handleSelectIndex: (index: any) => any, selectIndex: any, data: any) {
    return <div style={{ ...Styles.moreDetailBoxDetail, ...{ textAlign: 'left', overflowX: 'auto' } }} onClick={() => { handleSelectIndex(selectIndex); } }>
        {data && data.length > 0 && data.map((el: any) => { return (<div key={el.paxCabinType}>{`${el.paxCabinType ? el.paxCabinType : '-'}`}</div>); })}
    </div>;
}
 
function crewInfoGetDomain(handleSelectIndex: (index: any) => any, selectIndex: any, data: any) {

        {data && data.length > 0 && data.map((el: any) => { return (<div key={el.staffId}>{`${el.staffId ? el.staffId : '-'} | ${el.name ? el.name : '-'} | ${el.crewWorkPosition ? el.crewWorkPosition : '-'}`}</div>); })}
    </div>;
}
