import './App.css';
import Login from './screens/login';
import Home from './screens/home';
import FlightScoreDetails from './screens/flightScoreDetails';
import Alert from './atoms/alert';
import { BrowserRouter, Routes, Route } from "react-router-dom"
import { useEffect, useState } from "react";
import { useDispatch, useSelector } from 'react-redux';
import { loginAction } from './store/login/action';
import moment from 'moment'; 
import { timerAction } from './store/timer/action';
import { redirectToUnitedLogin,getAccessTokenApi,refershTokenApi } from './globals/OAUTH/oauthFunction';
import { stationTerminalAction } from './store/stationTerminal/action';
import { getTimingByTimeZone } from './globals/CommonTiming';

const App = ()=> {
 
  const[timeInterval,setTimeInterval]=useState(40); 
  // we need to change the below veriable value to "start" after adding the AOUTH data
  let isOAUTH = 'notStart';
  const dispatch = useDispatch()
  const isloggedin = useSelector((state: any) => state.UserData.isValidUserLoggedIn)
  const reducerTimerCounter = useSelector((state:any)=> state.timerData.timerCount)
  let storeTimeZone = useSelector((state:any)=> state.stationTerminalData.timeZone)
  let testStore = useSelector((state:any)=> state)

  const getReducerTimerCounterValue = ()=>{
    if(reducerTimerCounter===''){
      if(window.location.hostname==='gate-ai-1293274274.ap-south-1.elb.amazonaws.com' ||
       window.location.hostname === 'localhost'){
        return moment('09:45','HH:mm').subtract(timeInterval,'minutes').subtract(0,'day').format('HH:mm')
      }else{
        return moment(getTimingByTimeZone(storeTimeZone),'YYYY-MM-DDTHH:mm:SS').subtract(timeInterval,'minutes').format('HH:mm')
      }
    }else{
      // below thing for testing purpose
      // return moment('02:50','HH:mm').subtract(timeInterval,'minutes').subtract(0,'day').format('HH:mm'):
      return moment(reducerTimerCounter,'HH:mm').format('HH:mm')
    }
  }

  const timerCounter = getReducerTimerCounterValue()

    //OAUTH thing is start from here 
    useEffect(()=>{
      if(isOAUTH==='start'){
        console.log("localstorage", localStorage.getItem('tokenOIDC'))
        const search = window.location.search;
        const params = new URLSearchParams(search);
        const code = params.get('code');
        console.log("search Location", search, params, code)

        if (localStorage.getItem('tokenOIDC') === null && code === null&&localStorage.getItem('expires_timing') ===null) {
          localStorage.removeItem('codeVerifier');
          localStorage.removeItem('codeChallenge')
          redirectToUnitedLogin();
        } else if (code&&localStorage.getItem('expires_timing') ===null&&localStorage.getItem('tokenOIDC') === null) {
          getAccessTokenApi(code);
        }
      }
    },[])

    // for refersh the OAUTH token
    if(isOAUTH==='start'){
      setInterval(()=>{
        if(localStorage.getItem('expires_timing')!=null){
          const expiresTime = moment(localStorage.getItem('expires_timing'),'YYYY-MM-DDTHH:mm:SS')
          const currentTime = moment(moment().format('YYYY-MM-DDTHH:mm:SS'),'YYYY-MM-DDTHH:mm:SS')
          console.log('check refershTokenApi => 2',expiresTime.diff(currentTime,'minutes'))
          if(expiresTime.diff(currentTime,'minutes')===1){
            refershTokenApi()
          }
        }
      },60*1000)
    }
    //OAUTH thing is ended here 
  
  useEffect(()=>{
    if(reducerTimerCounter===''){
      dispatch(timerAction.storeTimer({time:timerCounter}))
    }
    setInterval(()=>{
      console.log('check timer value change =>',moment(timerCounter,'HH:mm').add(1,'minute').format('HH:mm'));
      (window.location.hostname!='gate-ai-1293274274.ap-south-1.elb.amazonaws.com' &&
      window.location.hostname != 'localhost')&&
        dispatch(timerAction.storeTimer({time:moment(moment(getTimingByTimeZone(storeTimeZone),'YYYY-MM-DDTHH:mm:SS').subtract(timeInterval+1,'minutes').format('HH:mm'),'HH:mm').add(1,'minute').format('HH:mm')}))
      // below thing for testing purpose
      // dispatch(timerAction.storeTimer({time:moment(timerCounter,'HH:mm').add(1,'minute').format('HH:mm')}))
    },60000)
},[reducerTimerCounter])


  const setTimeOnHoursSelection = (time:any)=>{
    console.log('check the setTimeOnHoursSelection => ',time)
    setTimeInterval(time);
  }

  useEffect(() => {
    if (localStorage.getItem('isValidUser') == 'true') {
      dispatch(loginAction.storeUserData({ accessToken: localStorage.getItem('accessToken') }))
      dispatch(stationTerminalAction.getStationAndTerminalList())
    }
    
  }, [])
  console.log('check the current page thing ===> ',localStorage.getItem('currentPage'))
  const currentPage = localStorage.getItem('currentPage')
  console.log('check the store value by testStore => ',testStore)
  const isFromVolar = false;
  return (
    <>
    <Alert />
    {/* <BrowserRouter basename='/iao/gateAI'> */}
    <BrowserRouter>
        <Routes>
            <Route path="/*" element={<Login />} />
            <Route path="/login" element={<Login />} />
            <Route path="/flightScoreDetails" element={<FlightScoreDetails />} />
            <Route path="/home" element={<Home currentPage={currentPage} timerCounter={timerCounter} setTimeOnHoursSelection={(el:any)=>{setTimeOnHoursSelection(el)}}/>} />

            {(localStorage.getItem('isValidUser') == 'true' || isloggedin)&&
            <Route path="/home" element={<Home currentPage={currentPage} timerCounter={timerCounter} setTimeOnHoursSelection={(el:any)=>{setTimeOnHoursSelection(el)}}/>} />}

            {(localStorage.getItem('isValidUser') == 'true' || isloggedin)&&
            <Route path="/flightScoreDetails" element={<FlightScoreDetails />} />}

            {(localStorage.getItem('isValidUser') == 'true' || isloggedin)&&
            <Route path="/login" element={<Login />} />}
        </Routes>
      </BrowserRouter>
      </>
  );
}
export default App;
