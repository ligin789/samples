export const getListStyles = (props:any) => {
    let {data,childComponent,type} = props;
    let key = childComponent?.props?.extraStyleKey ?? ''
   let value:any=""
   value=key=="visibilityColorChange"&&data?.["visibilityColor"]=="red"?"red":data?.["visibilityColor"]=="yellow"?"yellow":"#000000"
   value=key=="severity"&&data?.["severity"]=="HIGH"?"HIGH":data?.["severity"]=="MEDIUM"?"MEDIUM":data?.[key]
   if(childComponent?.type=="RsImage"&&key=="acknowledgeIconClicked"){
    if(data?.acknowledgeIconClicked==true){
        childComponent.key="acknowledgeIconClicked"
    }
    else{
        childComponent.key="acknowledgeIconNotClicked"
    }
   }
    const styles = childComponent?.props?.customStyles?.[value]?.style
    return styles ?? {}
}
