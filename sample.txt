import Constants from "../../global/Constants";
 
const initialState: any = {
  loading: false,
  data: null,
  error: null,
  isSearchByPatternError:false,
  isSearchByPatternComplete:false,
  showGridDataLoader:false,
  criticalWidgetDetails:{}
};
 
const HomeScreenReducer = (state = initialState, action: any) => {
  const newState = { ...state };
  switch (action.type) {
    case 'SAVE_REFERENCE_DATA':
      newState.data = action.payload;
      return newState
    case 'UPDATE_SEARCH_BY_PATTERN_COMPLETE_STATUS':
      newState.isSearchByPatternComplete = action.payload;
      return newState
    case 'UPDATE_SEARCH_BY_PATTERN_STATUS':
      newState.isSearchByPatternError = action.payload
      return newState
    case 'UPDATE_DATA_LOADER_STATUS':
    newState.showGridDataLoader = action.payload;
    return newState
    case 'SAVE_CRITICAL_WIDGET':
      newState.criticalWidgetDetails=action.payload
      return newState
    default:
      return state;
  }
};
 
export default HomeScreenReducer;
