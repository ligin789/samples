import React, { useState } from "react";
import DynamicRender from "../dynamicrenderer";
import { getJsonData } from "../../global/tenantConfig";

export const PilotCrewConxModal = (props: any) => {
    const {flightOb, col} = props;

    const getJsonForPilotCrewConx = (flightNo, column) => {
        if(flightNo.startsWith('1907')){
            return getJsonData("piloncrewconxJson")
        }
        if(flightNo.startsWith('223') && column === 'pilotconxob'){
            return getJsonData('piloncrewconxJson_223_pc_ob')
        }
        if(flightNo.startsWith('223') && column === 'pilotconxib'){
            return getJsonData('piloncrewconxJson_223_pc_ib')
        }
        if(flightNo.startsWith('223') && column === 'ccconxob'){
            return getJsonData('piloncrewconxJson_223_ccw_ob')
        }
        if(flightNo.startsWith('223') && column === 'ccconxib'){
            return getJsonData('piloncrewconxJson_223_ccw_ib')
        }   
    }
    const mapFunctions = {
        closeButtonClick: () => {
            
        },
    }
    return (
        <div style={{ position:'absolute' , zIndex: 1000, zoom: "var(--zoomPercentage)", right:0}}>
        <DynamicRender
            data={getJsonForPilotCrewConx(flightOb,col)}
            mappedFunctions={mapFunctions}
        />
    </div>
    )
}
export default PilotCrewConxModal
