import React, { createRef } from "react";
import "./dashboard.scss";
import NavBar from "../../components/features/navbar/navbar";
import DashboardRoute from "../../containers";
import Layout from "antd/lib/layout";
import { useHistory } from "react-router-dom";
export const historyRef = createRef();
 
const { Content, Sider } = Layout;
const Dashboard = ({ match ,
  //  access
  }) => {
  const history = useHistory();
  historyRef.current = history;
  
  return (
    <Layout>
      <Sider breakpoint="xl" collapsedWidth="0" className="dashboardnav">
        <NavBar url={match} 
        // access = {access}
         />
      </Sider>
      <Layout>
        <Content className="dashboard-content">
          <DashboardRoute url={match}  
          // access = {access} 
          />
        </Content>
      </Layout>
    </Layout>
  );
};
 
export default React.memo(Dashboard);


import Modal from "antd/lib/modal";
 
const info = ({ color = "#161D35", title, content, actions }) => {
  Modal.info({
    style: {
      backgroundColor: color,
    },
    title: title,
    content: content,
    onOk: actions,
  });
};
const success = ({ color = "#161D35", title, content, actions }) => {
  Modal.success({
    style: {
      backgroundColor: color,
    },
    title: title,
    content: content,
    onOk: actions,
  });
};
const errorAlert = ({ color = "#161D35", title, content, actions }) => {
  Modal.error({
    style: {
      backgroundColor: color,
    },
    title: title,
    content: content,
    onOk: actions,
  });
};
 
const warning = ({ color = "#161D35", title, content, actions }) => {
  Modal.warning({
    style: {
      backgroundColor: color,
    },
    title: title,
    content: content,
    onOk: actions,
  });
};
 
export { info, success, errorAlert, warning };
