import React, { useState } from "react"
import SupplementaryPopUpJson from "../../assets/jsons/supplementaryPopUp.json"
import DynamicRender from "../dynamicrenderer"
import { data } from "react-router-dom"
const SupplementaryPopUp = () => {
  const [text, setText]: any = useState("")
  const [jsonData, setJsonData] = useState(SupplementaryPopUpJson)
  const [modalVisible, setModalVisible]: any = useState(true)
  const mapFunctions: any = {
    supplementaryComment: (data: any) => {
      setText(data)
    },
    addButton: (data: any) => {
      let widgetData: any = SupplementaryPopUpJson
      let supplementaryInfo = { "supplementaryInfo": text }
      widgetData.widgetAttributes["#list"].push(supplementaryInfo)
      setJsonData(widgetData)
    },
    modalCloseButton: () => {
      setModalVisible(false)
    }
  }
  return (
    jsonData &&
    <div style={{ position: "absolute", zIndex: 10, left: "35%", top: "12%" }}>
      <DynamicRender
        data={jsonData}
        mainListIndex={1}
        mappedFunctions={mapFunctions}
      />
    </div>
  )
}
export default SupplementaryPopUp
