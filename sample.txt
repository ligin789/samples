const defaultChatReducerState = {
    taskTypeTabData: [],
    deptTypeTabData: [],
    focusFlightData: [],
    flightDetailData: [],
    gridData:[]
};
 
export default function FlightReducer(Istate = defaultChatReducerState, Iaction: any = null) {
    let newState = { ...state }
    switch (action.type) {
        case 'UPDATE_TASK_TYPE_TAB_DATA':
            newState.taskTypeTabData = action.payload
            return newState
        case 'UPDATE_DEPT_TYPE_TAB_DATA':
            newState.deptTypeTabData = action.payload;
            return newState
        case 'UPDATE_FOCUS_FLIGHT_DATA':
            newState.focusFlightData = action.payload
            return newState
        case 'UPDATE_FLIGHT_DETAIL_DATA':
            newState.flightDetailData = action.payload
            return newState
        case 'SAVE_GRID_DATA':
            newState.gridData = action.payload
            return newState
        default:
            return newState;
    }
}
